#STEP TWO: CALLS ##City of Fairfax Call Data for day, week, month, year, & type (Suspicious):

# Step 1
id.calls <- "1gRoL7lZlwm7sreA5F9WbPoH5su4n4iGS"
calls.temp <- read.csv(sprintf("https://docs.google.com/uc?id=%s&export=download", id.calls))

# Step 2
calls.temp$date <- as.Date(calls.temp$date)
calls.temp$dow <- weekdays(calls.temp$date)
calls.temp$day <- day(calls.temp$date)
calls.temp$month <- substr(calls.temp$date, 6, 7)

# Step 3
calls.type <- calls.temp %>%
  group_by(type) %>%
  summarise(count = n()) %>%
  mutate(PCT = round(count/sum(count)*100,2))
calls <- subset(calls.temp, calls.temp$type == 'SUSPICIOUS')

# Step 4
calls.day <- calls %>%
  group_by(date) %>%
  summarise(count = n())

first <- min(calls.day$date)
last <- max(calls.day$date)

calls.day <- calls.day %>% 
  complete(date = seq(ymd(first), 
                      ymd(last), 
                      "day")) %>%
  mutate(dow = wday(date, label = T, week_start = 1), 
         month.name = month(date, label = T, abbr = F),
         month.num = substr(date, 6,7),
         week = isoweek(date),
         day = day(date),
         year = year(date))

calls.day$count[is.na(calls.day$count)] = "0"
calls.day$count <- as.numeric(calls.day$count)

high <- as.numeric(max(calls.day$count))
low <- as.numeric(min(calls.day$count))
mean <- mean(calls.day$count)
sd <- sd(calls.day$count)

calls.day <- mutate(calls.day, 
                    week = case_when(month.name == "December" & week == 1 ~ 53,
                                     month.name == "January" & week %in% 52:53 ~ 0,
                                     TRUE ~ week),
                    category = cut(count, c(low-1, (mean-sd*3), (mean-sd*2), (mean-sd), mean, (mean+sd), (mean+sd*2), 
                                            (mean+sd*3), high+1)))

#ANALYSIS

#Month of year hotspots and coldspots, and what to expect during August 2023: The months of June (265), July (231), and August (235) yielded a lot of calls in 2013. Then August 2009 had 200 calls. Those were the only months that had upwards of 200 calls. For coldspots, February 2010 (57), November 2020 (74), February 2021 (76), January 2022 (61), and March 2022 (76) all had the lowest number of calls. If trends continue, August 2023 should see calls hover around the 120-155 mark. Notable exceptions for August were years 2020 (103) and 2021 (109), probably because of the pandemic.

#Day of week hotspots and coldspots, and what to expect next week: The hottest spots during the week days were (unsurprisingly) Friday, Saturday, and Sunday. The coldest spot happened on Monday. This really comes as no surprise as the weekend is when people are out enjoying themselves more and not at work. Hence, more likelihood to notice suspicious activities.I expect next weeks trends to continue the same way.

#Hour of day hotspots, coldspots, trends, and patterns: Hotspot hourly patterns were strongest between 0 (midnight) and 2 a.m., and 22 thru 23 (10 and 11 p.m.) hours. These times of day when most people are asleep, so criminals think this is the best time to strike. The coldest spots hourly are between 5 and 7 a.m. This is the time period when most people are waking up, drinking coffee, getting ready for work, letting pets out, etc. People stirring makes it less likely to get away with crime.
